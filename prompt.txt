## 📄 `src/cli.ts` — Complete Implementation

```typescript
#!/usr/bin/env bun

/**
 * DEV-JOB-FINDER CLI
 * A fresher-focused remote job aggregator with Rust/Solana superpowers
 *
 * Usage:
 *   bun run src/cli.ts                    # Interactive mode
 *   bun run src/cli.ts --github=pantha704 --roles=web3,backend --highlight=rust,solana
 *   npm install -g @pratham/job-finder && job-finder --experience=fresher
 */

import { parseArgs } from "bun";
import chalk from "chalk";
import figlet from "figlet";
import {
  select,
  checkbox,
  input,
  confirm,
  isCancel,
  cancel,
  outro,
  spinner,
} from "@inquirer/prompts";
import { runFullPipeline, type PipelineOptions } from "./index";
import { extractSkillsFromGitHub } from "./utils/github";
import { parseResumeKeywords } from "./utils/resume";
import { saveOutput, type OutputFormat } from "./utils/output";

// ─────────────────────────────────────────────────────────────
// CLI Banner & Branding
// ─────────────────────────────────────────────────────────────
const showBanner = () => {
  console.log(
    chalk.cyanBright(
      figlet.textSync("DEV-JOB-FINDER", {
        font: "ANSI Shadow",
        horizontalLayout: "default",
        verticalLayout: "default",
      })
    )
  );
  console.log(
    chalk.gray("   Find remote fresher jobs that actually want YOU\n")
  );
  console.log(
    chalk.dim(`   v${process.env.npm_package_version || "1.0.0"} • `) +
      chalk.green("Built by a fresher, for freshers") +
      chalk.dim(" • ") +
      chalk.blue("github.com/pantha704/job-finder") +
      "\n"
  );
};

// ─────────────────────────────────────────────────────────────
// Default Configuration (Fresher-Optimized)
// ─────────────────────────────────────────────────────────────
const DEFAULTS: PipelineOptions = {
  experience: "fresher", // "internship" | "fresher" | "junior"
  location: "remote-india", // "remote-global" | "remote-india" | "hybrid-kolkata"
  roles: ["web3", "backend"], // Default roles for Rust/Solana devs
  keywords: "", // Free-text search fallback
  highlight: ["rust", "solana", "typescript", "next.js"], // Auto-flag as 🔥 HIGH MATCH
  sources: ["internshala", "wellfound", "web3career", "ycombinator"], // Pre-checked for freshers
  format: "md" as OutputFormat, // "md" | "json" | "both"
  output: "job_opportunities.md",
  github: "", // Optional: extract skills from profile
  resume: "", // Optional: parse keywords from PDF
  showScore: true, // Display priority score [0-100]
  maxResults: 200, // Cap output to prevent overwhelm
  postedWithin: 14, // Days (default: 2 weeks)
  minStipend: 0, // INR/month (skip salary filter for v1)
  minSalary: 0, // INR/year
  appliedLog: "", // Path to JSON log to skip already-applied jobs
  verbose: false,
};

// ─────────────────────────────────────────────────────────────
// CLI Argument Parsing (Bun native)
// ─────────────────────────────────────────────────────────────
const parseCliArgs = () => {
  const { values, positionals } = parseArgs({
    args: Bun.argv,
    options: {
      github: { type: "string", short: "g" },
      resume: { type: "string", short: "r" },
      experience: { type: "string", short: "e" },
      location: { type: "string", short: "l" },
      roles: { type: "string", short: "R" },
      keywords: { type: "string", short: "k" },
      highlight: { type: "string", short: "H" },
      sources: { type: "string", short: "s" },
      format: { type: "string", short: "f" },
      output: { type: "string", short: "o" },
      "show-score": { type: "boolean", short: "S" },
      "max-results": { type: "string", short: "m" },
      "posted-within": { type: "string", short: "p" },
      "applied-log": { type: "string", short: "a" },
      verbose: { type: "boolean", short: "v" },
      help: { type: "boolean", short: "h" },
      version: { type: "boolean", short: "V" },
    },
    allowPositionals: true,
  });

  // Helper: parse comma-separated strings to arrays
  const toArray = (val?: string) =>
    val ? val.split(",").map((s) => s.trim().toLowerCase()) : undefined;

  return {
    values: {
      ...values,
      roles: toArray(values.roles),
      highlight: toArray(values.highlight),
      sources: toArray(values.sources),
      maxResults: values["max-results"] ? parseInt(values["max-results"]) : undefined,
      postedWithin: values["posted-within"] ? parseInt(values["posted-within"]) : undefined,
      showScore: values["show-score"] ?? DEFAULTS.showScore,
    },
    positionals,
  };
};

// ─────────────────────────────────────────────────────────────
// Interactive Prompt Flow (Progressive Disclosure)
// ─────────────────────────────────────────────────────────────
const runInteractiveMode = async (cliValues: Partial<PipelineOptions>) => {
  // Step 0: GitHub Profile (Skill Extraction)
  if (!cliValues.github) {
    const github = await input({
      message: "What's your GitHub username? (optional, for skill extraction)",
      placeholder: "e.g., pantha704",
      validate: (val) => val === "" || /^[a-zA-Z0-9-]+$/.test(val) || "Invalid username",
    });
    if (isCancel(github)) return cancel("Operation cancelled.");
    if (github) cliValues.github = github;
  }

  // Auto-extract skills if GitHub provided
  let extractedSkills: string[] = [];
  if (cliValues.github) {
    const loadSpinner = spinner({ text: "Fetching GitHub skills..." });
    loadSpinner.start();
    try {
      extractedSkills = await extractSkillsFromGitHub(cliValues.github);
      loadSpinner.succeed(chalk.green(`Found: ${extractedSkills.join(", ")}`));
    } catch {
      loadSpinner.warn("Could not fetch GitHub; using defaults");
      extractedSkills = DEFAULTS.highlight!;
    }
  }

  // Step 1: Experience Level (Strict fresher focus)
  if (!cliValues.experience) {
    cliValues.experience = (await select({
      message: "Select experience level:",
      choices: [
        { value: "fresher", label: "Fresher (0-1 years)", description: "Entry-level, new grads welcome" },
        { value: "internship", label: "Internship", description: "3-6 month stipend roles" },
        { value: "junior", label: "Junior (1-3 years)", description: "Some experience required" },
      ],
      default: "fresher",
    })) as PipelineOptions["experience"];
    if (isCancel(cliValues.experience)) return cancel("Operation cancelled.");
  }

  // Step 2: Location Preference
  if (!cliValues.location) {
    cliValues.location = (await select({
      message: "Location preference:",
      choices: [
        { value: "remote-india", label: "🇮🇳 Remote from India", description: "Companies hiring remotely in India" },
        { value: "remote-global", label: "🌍 Remote Global", description: "Worldwide remote (timezone flexible)" },
        { value: "hybrid-kolkata", label: "🏢 Hybrid/Kolkata", description: "Onsite or hybrid in Kolkata" },
      ],
      default: "remote-india",
    })) as PipelineOptions["location"];
    if (isCancel(cliValues.location)) return cancel("Operation cancelled.");
  }

  // Step 3: Target Roles (Multi-select with smart defaults)
  if (!cliValues.roles) {
    cliValues.roles = (await checkbox({
      message: "Select target roles (use arrow keys + space):",
      choices: [
        { value: "web3", label: "Web3/Blockchain", checked: true },
        { value: "backend", label: "Backend/Systems", checked: true },
        { value: "frontend", label: "Frontend", checked: false },
        { value: "devops", label: "DevOps/SRE", checked: false },
        { value: "ai-ml", label: "AI/ML Engineering", checked: false },
      ],
      validate: (vals) => vals.length > 0 || "Select at least one role",
    })) as PipelineOptions["roles"];
    if (isCancel(cliValues.roles)) return cancel("Operation cancelled.");
  }

  // Step 4: Tech Stack Highlighter (Auto-flag HIGH MATCH)
  if (!cliValues.highlight) {
    const defaultHighlights = extractedSkills.length > 0 ? extractedSkills : DEFAULTS.highlight!;
    cliValues.highlight = (await checkbox({
      message: `Highlight these tech stacks as ${chalk.redBright("🔥 HIGH MATCH")}:`,
      choices: [
        { value: "rust", label: "Rust", checked: defaultHighlights.includes("rust") },
        { value: "solana", label: "Solana", checked: defaultHighlights.includes("solana") },
        { value: "typescript", label: "TypeScript", checked: defaultHighlights.includes("typescript") },
        { value: "nextjs", label: "Next.js", checked: defaultHighlights.includes("next.js") },
        { value: "wasm", label: "WebAssembly", checked: false },
        { value: "anchor", label: "Anchor Framework", checked: false },
      ],
    })) as PipelineOptions["highlight"];
    if (isCancel(cliValues.highlight)) return cancel("Operation cancelled.");
  }

  // Step 5: Job Boards Selector (Smart defaults for freshers)
  if (!cliValues.sources) {
    cliValues.sources = (await checkbox({
      message: "Which job boards to scrape? (pre-checked for freshers):",
      choices: [
        { value: "internshala", label: "Internshala", description: "Best for Indian freshers ✓", checked: true },
        { value: "wellfound", label: "Wellfound (AngelList)", description: "Startup roles, entry-level ✓", checked: true },
        { value: "web3career", label: "Web3.career", description: "Crypto/Web3 roles ✓", checked: true },
        { value: "ycombinator", label: "YC Work at a Startup", description: "Global remote startups ✓", checked: true },
        { value: "remoterocketship", label: "Remote Rocketship", description: "Hidden remote gems", checked: false },
        { value: "linkedin", label: "LinkedIn", description: "⚠️ Requires auth, skip by default", checked: false },
      ],
      validate: (vals) => vals.length > 0 || "Select at least one source",
    })) as PipelineOptions["sources"];
    if (isCancel(cliValues.sources)) return cancel("Operation cancelled.");
  }

  // Step 6: Output Format
  if (!cliValues.format) {
    cliValues.format = (await select({
      message: "Output format:",
      choices: [
        { value: "md", label: "📝 Markdown (with checkboxes)", description: "Human-readable + track applications" },
        { value: "json", label: "🔧 JSON", description: "For automation/piping to other tools" },
        { value: "both", label: "✨ Both", description: "Generate .md and .json files" },
      ],
      default: "md",
    })) as OutputFormat;
    if (isCancel(cliValues.format)) return cancel("Operation cancelled.");
  }

  // Step 7: Output Path
  if (!cliValues.output) {
    const defaultPath = cliValues.format === "json" ? "job_opportunities.json" : "job_opportunities.md";
    cliValues.output = await input({
      message: "Save to:",
      default: defaultPath,
      validate: (val) => val.endsWith(".md") || val.endsWith(".json") || "File must end with .md or .json",
    });
    if (isCancel(cliValues.output)) return cancel("Operation cancelled.");
  }

  return cliValues as PipelineOptions;
};

// ─────────────────────────────────────────────────────────────
// Main Execution
// ─────────────────────────────────────────────────────────────
const main = async () => {
  showBanner();

  // Parse CLI args first (flags override interactive)
  const { values: cliValues } = parseCliArgs();

  // Handle help/version
  if (cliValues.help) {
    console.log(chalk.cyan("Usage:") + "\n" +
      "  job-finder [options]\n\n" +
      chalk.cyan("Options:") + "\n" +
      "  -g, --github <user>       GitHub username for skill extraction\n" +
      "  -e, --experience <level>  fresher | internship | junior (default: fresher)\n" +
      "  -l, --location <pref>     remote-india | remote-global | hybrid-kolkata\n" +
      "  -R, --roles <list>        Comma-separated: web3,backend,frontend,devops\n" +
      "  -H, --highlight <list>    Tech stacks to flag as 🔥 HIGH MATCH\n" +
      "  -s, --sources <list>      Job boards: internshala,wellfound,web3career,ycombinator\n" +
      "  -f, --format <fmt>        md | json | both (default: md)\n" +
      "  -o, --output <path>       Output file path\n" +
      "  -S, --show-score          Display priority score [0-100] for each job\n" +
      "  -v, --verbose             Enable debug logging\n" +
      "  -h, --help                Show this help message\n" +
      "  -V, --version             Show version\n"
    );
    process.exit(0);
  }

  if (cliValues.version) {
    console.log(`job-finder v${process.env.npm_package_version || "1.0.0"}`);
    process.exit(0);
  }

  // Merge CLI values with defaults
  const baseOptions: PipelineOptions = { ...DEFAULTS, ...cliValues };

  // Run interactive mode only if key params are missing
  const hasMinimalConfig = baseOptions.roles?.length && baseOptions.sources?.length;
  const finalOptions = !hasMinimalConfig
    ? await runInteractiveMode(baseOptions)
    : baseOptions;

  if (!finalOptions) return; // Cancelled

  // Optional: Parse resume for keyword boost
  if (finalOptions.resume) {
    const resumeSpinner = spinner({ text: "Parsing resume keywords..." });
    resumeSpinner.start();
    try {
      const resumeKeywords = await parseResumeKeywords(finalOptions.resume);
      finalOptions.keywords = [finalOptions.keywords, ...resumeKeywords].filter(Boolean).join(",");
      resumeSpinner.succeed(`Boosted with: ${resumeKeywords.join(", ")}`);
    } catch {
      resumeSpinner.warn("Could not parse resume; continuing with other filters");
    }
  }

  // Load applied log if provided (skip duplicates)
  let appliedJobs: Set<string> = new Set();
  if (finalOptions.appliedLog) {
    try {
      const log = JSON.parse(await Bun.file(finalOptions.appliedLog).text());
      appliedJobs = new Set(log.map((j: any) => `${j.company}-${j.role}`.toLowerCase()));
      console.log(chalk.dim(`ℹ Skipping ${appliedJobs.size} already-applied jobs\n`));
    } catch {
      console.warn(chalk.yellow("⚠ Could not load applied log; starting fresh"));
    }
  }

  // ─────────────────────────────────────────────────────────
  // EXECUTE PIPELINE
  // ─────────────────────────────────────────────────────────
  const mainSpinner = spinner({ text: "🔍 Starting job aggregation..." });
  mainSpinner.start();

  try {
    const results = await runFullPipeline({
      ...finalOptions,
      onJobFound: (job, count, total) => {
        // Real-time progress updates
        const score = finalOptions.showScore ? ` [${job.matchScore || 0}/100]` : "";
        const highlight = job.isHighMatch ? chalk.redBright(" 🔥 HIGH MATCH") : "";
        mainSpinner.text = `✓ [${count}/${total}] ${job.title} at ${job.company}${highlight}${score}`;
      },
      onSourceComplete: (source, count) => {
        console.log(chalk.green(`✓ Completed ${source}: ${count} jobs\n`));
      },
      skipApplied: appliedJobs, // Filter out already-applied
    });

    mainSpinner.succeed(chalk.green(`✨ Done! Found ${results.length} jobs.`));

    // Summary stats
    const highMatches = results.filter((j) => j.isHighMatch).length;
    const fresherFriendly = results.filter((j) => j.isFresherFriendly).length;
    console.log(chalk.dim("\n📊 Summary:"));
    console.log(`   • Total jobs: ${chalk.cyan(results.length)}`);
    console.log(`   • 🔥 HIGH MATCH: ${chalk.redBright(highMatches)}`);
    console.log(`   • Fresher-friendly: ${chalk.green(fresherFriendly)}`);
    console.log(`   • Sources used: ${chalk.blue(finalOptions.sources?.join(", ") || "N/A")}`);

    // Save output
    const savedPaths = await saveOutput(results, {
      path: finalOptions.output!,
      format: finalOptions.format!,
      includeTracker: finalOptions.format !== "json", // Checkbox tracker for MD only
    });
    console.log(chalk.dim(`\n💾 Saved to: ${savedPaths.join(", ")}`));

    // Pro tip
    if (highMatches > 0) {
      console.log(
        chalk.blueBright("\n💡 Pro tip:") +
          ` Apply to ${chalk.redBright("🔥 HIGH MATCH")} jobs within ${chalk.bold("48 hours")} for best response rate!\n`
      );
    }

    outro(chalk.green("Happy applying! 🚀"));

    // Exit with success
    process.exit(0);
  } catch (error) {
    mainSpinner.fail(chalk.red("Pipeline failed"));
    console.error(chalk.red("\n❌ Error:"), error instanceof Error ? error.message : error);
    if (finalOptions.verbose) {
      console.error(chalk.dim("\nStack trace:"), error);
    }
    console.log(chalk.dim("\n💡 Try: job-finder --help for usage tips"));
    process.exit(1);
  }
};

// ─────────────────────────────────────────────────────────────
// Entry Point
// ─────────────────────────────────────────────────────────────
main().catch((err) => {
  console.error(chalk.red("Unhandled error:"), err);
  process.exit(1);
});
```

```typescript
// src/filters/location.ts
// Sophisticated location matching logic for Indian job market

import { Location, isLocationInIndia, searchLocations } from '../config/locations';
import { PipelineOptions, FilteredJob, LocationScope } from '../types/options';

// ─────────────────────────────────────────────────────────
// LOCATION NORMALIZATION
// ─────────────────────────────────────────────────────────

export const normalizeLocation = (raw: string, options: PipelineOptions): {
  normalized: string;
  scope: LocationScope;
  locationId?: string;
} => {
  const lower = raw.toLowerCase().trim();

  // Remote patterns
  if (lower.includes('remote') && lower.includes('india')) {
    return { normalized: 'remote-india', scope: 'remote-india' };
  }
  if (lower.includes('remote') && (lower.includes('global') || lower.includes('worldwide'))) {
    return { normalized: 'remote-global', scope: 'remote-global' };
  }
  if (lower.includes('remote') && !lower.includes('india')) {
    // Default to remote-global but flag for verification
    return { normalized: 'remote-global', scope: 'remote-global' };
  }

  // Hybrid/Onsite patterns
  if (lower.includes('hybrid') || lower.includes('flexible')) {
    return { normalized: 'hybrid', scope: 'hybrid' };
  }

  // Try to match against known Indian locations
  const matches = searchLocations(raw);
  if (matches.length > 0) {
    const bestMatch = matches[0]; // Most relevant match
    return {
      normalized: bestMatch.id,
      scope: options.locationScope === 'remote-india' ? 'remote-india' : 'onsite',
      locationId: bestMatch.id,
    };
  }

  // Fallback: detect if it's in India via keyword matching
  if (isLocationInIndia(raw)) {
    return { normalized: 'india-unspecified', scope: 'onsite' };
  }

  // Default: assume global/unknown
  return { normalized: 'unknown', scope: 'any' };
};

// ─────────────────────────────────────────────────────────
// LOCATION FILTERING LOGIC
// ─────────────────────────────────────────────────────────

export const matchesLocationFilter = (
  job: FilteredJob,
  options: PipelineOptions
): boolean => {
  const { locationScope, locations, locationRadius, remoteFromIndia } = options;

  // If no location filter, accept all
  if (!locationScope && !locations?.length && remoteFromIndia === undefined) {
    return true;
  }

  const jobScope = job.location.scope;
  const jobLocId = job.location.locationId;

  // ─────────────────────────────────────────────────────
  // Remote-from-India explicit filter
  // ─────────────────────────────────────────────────────
  if (remoteFromIndia === true) {
    return (
      jobScope === 'remote-india' ||
      (jobScope === 'remote-global' && job.location.raw.toLowerCase().includes('india'))
    );
  }

  // ─────────────────────────────────────────────────────
  // Location scope filter
  // ─────────────────────────────────────────────────────
  if (locationScope && locationScope !== 'any') {
    if (locationScope === 'remote-india') {
      // Accept: remote-india OR remote-global that mentions India
      if (jobScope === 'remote-india') return true;
      if (jobScope === 'remote-global' && job.location.raw.toLowerCase().includes('india')) {
        return true;
      }
      return false;
    }
    if (jobScope !== locationScope) {
      return false;
    }
  }

  // ─────────────────────────────────────────────────────
  // Specific location IDs filter (for onsite/hybrid)
  // ─────────────────────────────────────────────────────
  if (locations?.length && jobLocId) {
    // Exact match
    if (locations.includes(jobLocId)) return true;

    // Parent state/UT match (if job is in Kolkata, and user selected West Bengal)
    const jobLocation = searchLocations(jobLocId)[0];
    if (jobLocation?.state) {
      const stateId = `IN-${jobLocation.state.substring(0, 2).toUpperCase()}`;
      if (locations.includes(stateId)) return true;
    }
    if (jobLocation?.ut) {
      const utId = `IN-${jobLocation.ut.substring(0, 2).toUpperCase()}`;
      if (locations.includes(utId)) return true;
    }

    // Radius match for hybrid (simplified: same state = within 100km)
    if (locationRadius && jobScope === 'hybrid') {
      // In production: use geocoordinates + haversine formula
      // For now: same state = match
      const userLocations = locations.map((id) => searchLocations(id)[0]).filter(Boolean);
      const userStates = new Set(userLocations.map((l: Location) => l.state || l.ut));
      if (jobLocation?.state && userStates.has(jobLocation.state)) return true;
      if (jobLocation?.ut && userStates.has(jobLocation.ut)) return true;
    }

    return false;
  }

  return true;
};

// ─────────────────────────────────────────────────────────
// LOCATION SCORING (for priority ranking)
// ─────────────────────────────────────────────────────────

export const scoreLocationMatch = (
  job: FilteredJob,
  options: PipelineOptions
): number => {
  let score = 0;

  // Remote-from-India bonus (high value for Indian freshers)
  if (options.remoteFromIndia && job.location.scope === 'remote-india') {
    score += 25;
  }

  // Exact location match bonus
  if (options.locations?.includes(job.location.locationId || '')) {
    score += 20;
  }

  // State/UT match bonus
  if (job.location.locationId) {
    const jobLoc = searchLocations(job.location.locationId)[0];
    const matchedStates = options.locations?.filter((id) => {
      const loc = searchLocations(id)[0];
      return loc?.type === 'state' && loc.name === jobLoc?.state;
    }).length || 0;
    if (matchedStates > 0) score += 10;
  }

  // Remote-global with India mention (good but not perfect)
  if (job.location.scope === 'remote-global' && job.location.raw.toLowerCase().includes('india')) {
    score += 15;
  }

  // Hybrid in preferred city (moderate value)
  if (job.location.scope === 'hybrid' && options.locations?.includes(job.location.locationId || '')) {
    score += 12;
  }

  return Math.min(score, 30); // Cap location score contribution
};
```
